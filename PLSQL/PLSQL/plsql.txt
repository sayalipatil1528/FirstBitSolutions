create procedure updateSalary(nm varchar(20))
begin
	declare sal int;
	select salary into sal from employee where name=nm;
	if(sal>=25000)then
	update employee set salary =salary+(salary*0.1) where name=nm;
	else
	update employee set salary=salary+(salary*0.2) where name=nm;
	end if;
end

create function demo() returns varchar(20)
deterministic
begin
	call updateSalary("sneha");
	return "done";
end
